427     *

429     * only trusted when the request originates from a known trusted proxy IP/CIDR.

436    private static function get_client_ip(): string {
437        $remote_addr = $_SERVER['REMOTE_ADDR'] ?? '';
438
439        if ( ! filter_var( $remote_addr, FILTER_VALIDATE_IP ) ) {

452                    }
453                }
454            }
455        }

457        return $remote_addr;

462     *
463     * @param string $ip The REMOTE_ADDR to check.
464     * @return bool
465     */
466    private static function is_trusted_proxy( string $ip ): bool {
467        // Loopback is always trusted (local reverse proxies, Docker internal networking).
468        if ( in_array( $ip, [ '127.0.0.1', '::1' ], true ) ) {
469            return true;
470        }
471

473            return false;
474        }
475
476        $trusted = array_filter( array_map( 'trim', explode( ',', LTMS_TRUSTED_PROXY_IPS ) ) );
477
478        foreach ( $trusted as $entry ) {
479            if ( str_contains( $entry, '/' ) ) {
480                if ( self::ip_in_cidr( $ip, $entry ) ) {
481                    return true;
482                }
483            } elseif ( $ip === $entry ) {
484                return true;
485            }
486        }

488        return false;

492     * Checks whether an IPv4 address falls within a CIDR range.
493     *
494     * @param string $ip   IPv4 address to test.
495     * @param string $cidr CIDR notation (e.g. "10.0.0.0/8").
496     * @return bool
497     */
498    private static function ip_in_cidr( string $ip, string $cidr ): bool {
499        $parts = explode( '/', $cidr, 2 );
500        if ( count( $parts ) !== 2 ) {
501            return false;
502        }
503
504        [ $subnet, $prefix ] = $parts;
505        $prefix      = (int) $prefix;
506        $ip_long     = ip2long( $ip );
507        $subnet_long = ip2long( $subnet );
508
509        if ( false === $ip_long || false === $subnet_long || $prefix < 0 || $prefix > 32 ) {
510            return false;
511        }
512
513        $mask = $prefix === 0 ? 0 : ( ~0 << ( 32 - $prefix ) );
514        return ( $ip_long & $mask ) === ( $subnet_long & $mask );
515    }
