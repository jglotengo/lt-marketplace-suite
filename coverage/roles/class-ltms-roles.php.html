242     * Filtro de capacidades dinámicas: añade restricciones contextuales.
243     * Ej: Un vendedor solo puede gestionar sus propios productos.
244     *
245     * @param bool[]   $allcaps Todas las capacidades del usuario.
246     * @param string[] $caps    Capacidades solicitadas.
247     * @param array    $args    Argumentos (0=cap, 1=user_id, 2=object_id).
248     * @param \WP_User $user    Usuario actual.
249     * @return bool[]
250     */
251    public static function dynamic_capabilities( array $allcaps, array $caps, array $args, \WP_User $user ): array {
252        // Si no hay args[2] (object_id), no hay restricción adicional
253        if ( empty( $args[2] ) ) {
254            return $allcaps;
255        }
256
257        $capability = $args[0] ?? '';
258        $object_id  = (int) $args[2];
259
260        // Restricción: Vendedor (estándar y premium) solo puede editar sus propios productos
261        if (
262            $capability === 'edit_post' &&
263            ( in_array( 'ltms_vendor', (array) $user->roles, true ) ||
264              in_array( 'ltms_vendor_premium', (array) $user->roles, true ) ) &&
265            $object_id > 0
266        ) {
267            $post = get_post( $object_id );
268            if ( $post && $post->post_author != $user->ID ) {
269                $allcaps['edit_post'] = false;
270            }
271        }
272
273        return $allcaps;
274    }
